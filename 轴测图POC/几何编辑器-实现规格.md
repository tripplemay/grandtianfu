## ② 房间→墙体 推导算法 `derive(G)`(纯函数·精确伪码)

`EPS=meta.eps`;录入先吸附 `GRID=meta.grid`。无副作用,可单测。

```text
derive(G):
  rooms = snap_rects(G.rooms, GRID)
  validate_no_cross_space_overlap(rooms)          # D1:跨 space 净矩形重叠 → ERROR
  t = G.meta.wall_thickness_mm; mm = G.meta.mm_per_px

  # ---- occupancy(点归属;D1:去掉最小面积分支)----
  occupancy(px, py):
     hits = [r for r in rooms if r.rect contains (px,py) within EPS]
     if not hits: return VOID                      # category='void', space=None
     spaces = unique(r.space for r in hits)
     if len(spaces) > 1: raise Conflict(px,py,spaces)   # 跨 space 重叠点 → ERROR(不裁决)
     r = hits[0]; return (r.space, cat(r.space))

  # ---- classify(融合;D3 补 outdoor 四规则;D4 户分隔档)----
  classify(A, B):                                  # A=side−, B=side+
     if A.space and B.space and A.space==B.space: return None          # 同 space → 开放无墙(难点2)
     ia,ib = (A.cat=='interior'),(B.cat=='interior')
     oa,ob = (A.cat=='outdoor'), (B.cat=='outdoor')
     sa,sb = (A.cat=='shared'),  (B.cat=='shared')
     va,vb = (A.cat=='void'),    (B.cat=='void')
     if va and vb: return None
     # outdoor 显式四规则(D3):
     if oa and ob: return None                                         # 露台↔露台 开放
     if (oa and ib) or (ob and ia): return 'outdoor'                   # 室内↔露台(带玻璃) 240
     if (oa and vb) or (ob and va): return 'exterior'                  # 露台↔void 栏板/外缘(必须显式,防消失)
     if (oa and sb) or (ob and sa): return 'exterior'                  # 露台↔公共
     # interior 系:
     if ia and ib: return 'interior'                                   # 异室内 space → 内墙 140(难点2)
     if (ia and vb) or (ib and va): return 'exterior'                  # 室内↔void → 真外墙 240(难点1)
     if (ia and sb) or (ib and sa): return 'demarcation'              # 室内↔公共 → 户分隔 200(finding6)
     # shared 系:
     if sa and sb: return None                                         # 同公共(异 space 时仍 None,装饰墙走 free_walls)
     if (sa and vb) or (sb and va): return 'public'                    # 公共↔void → 虚线外缘(难点5)
     return None

  # ---- cell decomposition:逐线切最小子段(难点1/2)----
  raw = []
  for axis in ('h','v'):
     lines  = sorted(unique( 该向房间边坐标 ))                          # h:y边 / v:x边
     breaks = sorted(unique( 正交向所有房间边坐标 ))
     for c in lines:
        for (lo,hi) in pairs(breaks):
           if hi-lo < EPS: continue
           m = (lo+hi)/2
           A = occupancy(sample_minus(c,m,axis)); B = occupancy(sample_plus(c,m,axis))
           role = classify(A,B)
           if role: raw.append(Wall(axis,c,lo,hi,role))

  walls = merge_collinear(raw)                     # 吃掉 T/十字假断点(同 axis,at,role 且端点 gap<EPS)

  # ---- 并入自由墙,再合一次(难点4)----
  for fw in G.free_walls:
     walls.append(Wall(fw.axis, fw.at, fw.span, fw.role, style=fw.get('style')))
  walls = merge_collinear(walls)

  # ---- 开洞(难点3;顺序:先 merge 后 subtract,D 对南墙跨 x=1005 T 接点显式钉死 finding18)----
  doors=[]; windows=[]; conflicts=[]; warns=[]
  for op in G.openings:
     if op.cut:
        host = find_continuous_wall(walls, op.wall)            # span ⊆ 一条连续墙?
        if host is None or not span_subset(op, host, EPS):
           conflicts.append(Error("opening %s 不落任何连续派生墙" % op.id))   # D12 → ERROR
           continue
        walls = subtract_interval(walls, op.wall.axis, op.wall.at, op.wall.span)  # 1段→0/1/2段
     if   op.kind=='door':   doors.append(build_door(op))      # ③
     elif op.kind=='window': windows.append(window_rect(op, walls, t, mm))  # at=墙中线;宽=墙厚(finding17)
     # passage:只切墙不出扇

  # ---- 赋厚度/样式/挤出语义 ----
  for w in walls:
     w.ext   = (w.role in ('exterior','outdoor'))
     w.style = w.style or {'exterior':'solid','demarcation':'solid','interior':'solid',
                           'outdoor':'solid','thin':'thin','public':'dashed'}[w.role]
     w.T_px  = t[w.role]/mm                                    # 24/20/14/24/6/6
     w.lowz  = (w.role in ('thin','public'))                   # D9:轴测不挤整高
  return { walls=[(ax,ay,bx,by,w.ext,w.style,w.lowz)...], doors, windows,
           dims=gen_dims(G,rooms,walls), conflicts, warns }
```

door_frame(op):

## ④ 自动尺寸标注 `gen_dims`(刻度由"抵达立面的墙"而非全体房间边 finding20/21)

```text
gen_dims(G, rooms, walls):
  FP = footprint_polygon( union of rects where cat in {interior,outdoor,shared} )   # 整块楼板(finding21)
  for side in G.dims.sides:                       # 默认 [top,left]
     edge = boundary_polyline(FP, side)           # 沿该立面外轮廓走线
     ticks = sorted(unique(
        [拐点.coord for 拐点 in edge.corners] +    # (a) 轮廓拐点
        [w.at for w in walls
            if w.axis ⟂ side
            and w 端点 span 触及该立面(到 edge 的法向距 < EPS)] ))   # (b) 真正交到该立面的正交墙
        − G.dims.exclude_coords
     for (a,b) in pairs(ticks):
        if segment(a,b) lies on FP boundary(side):    # L 形凹口处断链(难点6)
           emit tick(a),tick(b),seg(a,b,text=(b-a)*mm)
     emit_chain(side, offset=offsets_px[side])
  emit_total(每侧 FP 总跨, 再外移一档)
  apply(G.dims.overrides)
```
